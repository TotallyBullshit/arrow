xUniqueOf

Usage

xUniqueOf(coll)


xUniqueOf...(...)

Arguments

coll
a collection. The collection to return unique elements from.
...
see above.

Value

A list.

Description

Return the unique elements in a collection.

Corner Cases

Returns the empty list if colls is length-zero.

Examples


# 1. Find the smallest unique number in a list.

lowest <-
x_( list(1, 1, 3, 2, 2, 2, 5, 4, 9, 9) ) $
xUniqueOf() $
x_Tap(min %of% xAsDouble)

# 2. CE is a string a pangram?
#    does a string contain each letter.

is_pangram <- line := {

	x_(line) $
	xToChars() $
	xMap(tolower) $
	xUniqueOf() $
	xSelect(
		xPartial...(xIsMember, coll = letters)
	) $
	x_LenOf() == 26
}

is_pangram('A quick brown fox jumps over the lazy dog')



[1] TRUE




# True

is_pangram('not pangramic thank you mam.')



[1] FALSE




# False
